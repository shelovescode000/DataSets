details = cab_holiday_fulldata['Income'].describe()
lowerQ = details['25%']
mean = details["mean"]
upperQ = details['75%']

# Generate result using pandas 
result = [] 
for value in cab_holiday_fulldata["Income"]: 
    if value < lowerQ: 
        result.append("low") 
    elif value<mean and value > lowerQ:
        result.append("middle") 
    else: 
        result.append("high")

#Customer retention over the years
g = cab_holiday_fulldata.groupby(cab_holiday_fulldata.Company)
Yellow_cab = g.get_group('Yellow Cab')
Pink_cab = g.get_group('Pink Cab')

yellow_year = Yellow_cab.groupby(['year','Customer ID'])['Transaction ID'].size().unstack()
pink_year = Pink_cab.groupby(['year','Customer ID'])['Transaction ID'].size().unstack()

yr = (yellow_year.sum(axis=0,skipna=True))/3
pr = (pink_year.sum(axis=0,skipna=True))/3

bins = [0,10,40]
labels = ['bad', 'good']
Yellow_cab['retention'] = pd.cut(yr, bins, labels = labels,include_lowest = True)
Pink_cab['retention'] = pd.cut(pr, bins, labels = labels,include_lowest = True)
